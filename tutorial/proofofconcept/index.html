Descirption of Componenet

My component is two different modal buttons that when pressed, open their repsective form. Each form takes two different input values along with an email address. It validates the user's inputs, then alerts the user an alert depending on which form they are inputting to. It then saves the inputted information to local storage and console logs the inputs

This is the java script that makes the content of my index page. Click Here to Try It

            
                function makeContent(id, object) {
                var tag = document.createElement("div");
                tag.setAttribute('id', 'bodyContent');
                var head = document.createElement("h2");
                head.innerText = object.frmtitle;
                tag.appendChild(head);
                var button = document.createElement("button");
                button.setAttribute('id', object.btnId);
                button.setAttribute('class', 'btnClass');
                button.innerText = object.btnInnerText;
                tag.appendChild(button);
                var mod1 = document.createElement("div");
                mod1.setAttribute('id', object.modId);
                mod1.setAttribute('class', 'modal');
                tag.appendChild(mod1);

                var element = document.getElementById(id);
                element.appendChild(tag);

                function formContent() {

                modDiv.appendChild(mod1con);
                };

                var modal = document.getElementById("myModal");
                var btn = document.getElementById("myBtn");
                var span = document.getElementsByClassName("close")[0];

                btn.onclick = function () {
                modal.style.display = "block";
                };

                span.onclick = function () {
                modal.style.display = "none";
                };

This is my reusable function that creates inputs with specified text value, class, id, placeholder, and type Click Here to Try It

            
                function createInput(text, theClass, type, thePlaceholder, theId) {
                var input = document.createElement("input");
                type = "input";
                input.setAttribute("type", type);
                input.className = theClass;
                input.textContent = text;
                input.placeholder = thePlaceholder;
                input.id = theId;
                return input;
                }
